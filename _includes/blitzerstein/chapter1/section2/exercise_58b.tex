\begin{exercise}{58b}
    Find the probability that the inspector will now have to test at least 10 widgets.
\end{exercise}

\begin{proof}
    We apply the same line of thought and obtain

    $$p = \frac{ {9 \choose 3} \cdot 3! \cdot 9!}{12!} = \frac{34}{55} \approx 0.61818$$

\end{proof}
